# 中间层上层结构体定义

```c++
typedef struct rr_ msg_lidar_point_data
{
    float x;
    float y; 
    float z;
    uint8_t intensity; 
    uint6_t ring;
    float timestamp_offset; 
    uint8_t tag;
}rr_msg_lidar_point_data_t;
#define LIDAR_MAX_POINT_COUNT (900 * 96 +20* 96)
typedef struct rr_msg_lidar_pointcloud_data
{
    uint64_t    ap_timestamp;   
    uint32_t    point_num;  
    uint32_t    height; 
    uint32_t    width;  
    bool        is_dense;   
    double      lidar_timestamp;    
    uint32_t    seq;    
    char        frame_id[64];   
    rr_msg_lidar_point_data_t points[LIDAR_MAX_POINT_COUNT];
} rr_msg_lidar_pointcloud_data_t;

typedef struct rr_msg_lidar_imu_data {
    double  timestamp;  
    float   orientation_x;  
    float   orientation_y;  
    float   orientation_z;  
    float   orientation_wi  
    float   angular_velocity_x;
    float   angular_velocity_y;
    float   angular_velocity_z; 
    float   linear_acceleration_x;  
    float   linear_acceleration_y;  
    float   linear_acceleration_z;  
}rr_msg_lidar_imu_data_t;
```
